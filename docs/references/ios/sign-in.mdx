# Create a Sign In

<Tabs type="framework" items={["SwiftUI", "UIKit"]}>

<Tab>
<CodeBlockTabs options={["SignInView.swift"]}>

```swift filename="SignInView.swift"
import SwiftUI
import ClerkSDK

struct SignInView: View {
    @ObservedObject private var clerk = Clerk.shared
    @State private var emailAddress: String = ""
    @State private var password: String = ""
    
    var body: some View {
        VStack {
            if let user = clerk.user {
                Text(user.id)
            } else {
                TextField("Email", text: $emailAddress)
                SecureField("Password", text: $password)
                Button("Sign In", action: {
                    Task {
                        try? await clerk.client?.createSignIn(strategy: .identifier(emailAddress, password: password))
                    }
                })
            }
        }
    }
}
```

```swift
import UIKit
import ClerkSDK

class SignInViewController: UIViewController {
    @IBOutlet weak var userIdLabel: UILabel!
    @IBOutlet weak var emailTextField: UITextField!
    @IBOutlet weak var passwordTextField: UITextField!
        
    @IBAction func signIn(_ sender: Any) {
        guard let emailAddress = emailTextField.text, let password = passwordTextField.text else {
            return
        }
        
        Task {
            try? await Clerk.shared.client?.createSignIn(strategy: .identifier(emailAddress, password: password))
            userIdLabel.text = Clerk.shared.user?.id
        }
    }
}
```
