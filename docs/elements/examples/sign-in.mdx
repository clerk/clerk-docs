# Sign-in

## Username/password

A basic username/password sign-in screen. Since there is no second factor, only `<Step name="start">` will suffice.

```tsx {{ filename: '/app/sign-in/[[...sign-in]]/page.tsx', collapsible: true }}
'use client'

import * as Clerk from '@clerk/elements/common'
import * as SignIn from '@clerk/elements/sign-in'

export function SignInPage() {
  return (
    <SignIn.Root>
      <SignIn.Step name="start">
        <Clerk.GlobalError />
        <Clerk.Field name="identifier">
          <Clerk.Label>Username</Clerk.Label>
          <Clerk.Input type="text" required />
          <Clerk.FieldError />
        </Clerk.Field>
        <Clerk.Field name="password">
          <Clerk.Label>Password</Clerk.Label>
          <Clerk.Input type="password" required />
          <Clerk.FieldError />
        </Clerk.Field>
        <SignIn.Action submit>Sign In</SignIn.Action>
      </SignIn.Step>
    </SignIn.Root>
  )
}
```

## Google OAuth

Users can only sign in with Google. Note: Depending on your settings, you may require additional `Steps`.

```tsx {{ filename: '/app/sign-in/[[...sign-in]]/page.tsx', collapsible: true }}
'use client';

import * as Clerk from '@clerk/elements/common'
import * as SignIn from '@clerk/elements/sign-in'

export default function SignInPage() {
  return (
	  <SignIn.Root>
	    <SignIn.Step name="start">
	      <Clerk.GlobalError />
	      <Clerk.Connection name="google">
	        <Clerk.Icon /> Login with Google
	      </Clerk.Connection>
	    </SignIn.Step>
	  </SignIn.Root>
	)
}
```

## Multi-Factor Auth

A user has to provide their password and also a second factor - in this case an SMS code sent to their phone.


```tsx {{ filename: '/app/sign-in/[[...sign-in]]/page.tsx', collapsible: true }}
'use client';

import * as Clerk from "@clerk/elements/common"
import * as SignIn from "@clerk/elements/sign-in"

export default function SignInPage() {
  return (
	  <SignIn.Root>
	    <SignIn.Step name="start">
	      <Clerk.GlobalError />
	      <Clerk.Field name="identifier">
	        <Clerk.Label>Username</Clerk.Label>
	        <Clerk.Input type="text" required />
	        <Clerk.FieldError />
	      </Clerk.Field>
	      <Clerk.Field name="password">
	        <Clerk.Label>Password</Clerk.Label>
	        <Clerk.Input type="password" required />
	        <Clerk.FieldError />
	      </Clerk.Field>
	      <SignIn.Action submit>Sign In</SignIn.Action>
	    </Step>
	    <SignIn.Step name="verifications">
	      <Clerk.GlobalError />
	      <SignIn.Strategy name="phone_code">
	        <Clerk.Field name="code">
	          <Clerk.Label>Phone code</Clerk.Label>
	          <Clerk.Input type="otp" required />
	          <Clerk.FieldError />
	        </Clerk.Field>
	        <SignIn.Action submit>Continue</SignIn.Action>
	      </SignIn.Strategy>
	    </SignIn.Step>
	  </SignIn.Root>
	)
}
```

## Sign in with Google or Email + MFA

A user can sign in with Google or by email. They will then need to provide both an email code and a phone code.

```tsx {{ filename: '/app/sign-in/[[...sign-in]]/page.tsx', collapsible: true }}
'use client';

import * as Clerk from "@clerk/elements/common"
import * as SignIn from "@clerk/elements/sign-in"

export default function SignInPage() {
  return (
	  <SignIn.Root>
	    <SignIn.Step name="start">
	      <Clerk.GlobalError />
	      <Clerk.Connection name="google">
	        <Clerk.Icon /> Login with Google
	      </Clerk.Connection>
	      <Clerk.Field name="identifier">
	        <Clerk.Label>Email</Clerk.Label>
	        <Clerk.Input type="email" required />
	        <Clerk.FieldError />
	      </Clerk.Field>
	      <SignIn.Action submit>Sign In</SignIn.Action>
	    </SignIn.Step>
	    <SignIn.Step name="verifications">
	      <Clerk.GlobalError />
	      <SignIn.Strategy name="email_code">
	        <Clerk.Field name="code">
	          <Clerk.Label>Email code</Clerk.Label>
	          <Clerk.Input type="otp" required />
	          <Clerk.FieldError />
	        </Clerk.Field>
	        <Action submit>Continue</Action>
	      </SignIn.Strategy>
	      <SignIn.Strategy name="phone_code">
	        <Clerk.Field name="code">
	          <Clerk.Label>Phone code</Clerk.Label>
	          <Clerk.Input type="otp" required />
	          <Clerk.FieldError />
	        </Clerk.Field>
	        <SignIn.Action submit>Login</SignIn.Action>
	      </SignIn.Strategy>
	    </SignIn.Step>
	  </SignIn.Root>
	)
}
```
