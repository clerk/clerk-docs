The `customizing-auth-ui` skill helps AI agents build fully custom sign-in and sign-up experiences using Clerk's hooks.

## Installation

```bash
npx skills add clerk/skills --skill customizing-auth-ui
```

## When to customize

| Use pre-built components | Use custom UI |
| --- | --- |
| Quick setup | Specific design requirements |
| Standard auth flows | Multi-step onboarding |
| `appearance` prop is sufficient | Full brand control |

## Example prompts

- "Build a custom sign-in form with email and password"
- "Create a sign-up page with email verification"
- "Add Google and GitHub OAuth buttons"
- "Build a multi-step onboarding flow"

## Key hooks

```tsx
import { useSignIn, useSignUp } from '@clerk/nextjs'

const { signIn, setActive, isLoaded } = useSignIn()
const { signUp, setActive, isLoaded } = useSignUp()
```

## Common patterns

### Custom sign-in form

```tsx {{ filename: 'app/sign-in/[[...sign-in]]/page.tsx' }}
'use client'

import { useSignIn } from '@clerk/nextjs'
import { useRouter } from 'next/navigation'
import { useState } from 'react'

export default function SignInPage() {
  const { signIn, setActive, isLoaded } = useSignIn()
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const router = useRouter()

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (!isLoaded) return

    const result = await signIn.create({
      identifier: email,
      password,
    })

    if (result.status === 'complete') {
      await setActive({ session: result.createdSessionId })
      router.push('/dashboard')
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="Email"
      />
      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="Password"
      />
      <button type="submit">Sign In</button>
    </form>
  )
}
```

### OAuth buttons

```tsx {{ filename: 'components/OAuthButtons.tsx' }}
'use client'

import { useSignIn } from '@clerk/nextjs'

export function OAuthButtons() {
  const { signIn, isLoaded } = useSignIn()

  const signInWith = (strategy: 'oauth_google' | 'oauth_github') => {
    if (!isLoaded) return
    signIn.authenticateWithRedirect({
      strategy,
      redirectUrl: '/sso-callback',
      redirectUrlComplete: '/dashboard',
    })
  }

  return (
    <div>
      <button onClick={() => signInWith('oauth_google')}>
        Continue with Google
      </button>
      <button onClick={() => signInWith('oauth_github')}>
        Continue with GitHub
      </button>
    </div>
  )
}
```

## Common pitfalls

- **Check `isLoaded` before using hooks** - `signIn` and `signUp` are undefined until loaded
- **Call `setActive()` after sign-in** - the session won't activate without it
- **Handle all statuses** - `complete`, `needs_first_factor`, `needs_second_factor`
- **Import the error helper** - `import { isClerkAPIResponseError } from '@clerk/nextjs/errors'`
