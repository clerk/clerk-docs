# `<ClerkLoading>`

---

The `<ClerkLoading>` renders its children while Clerk is loading, and is helpful for showing a custom loading state.

## Usage

```jsx copy filename="app.tsx"
import React, { useEffect } from 'react';
import { MultisessionAppSupport, ClerkProvider } from '@clerk/nextjs';

declare global {
  interface Window {
    Clerk: any;
  }
}

function App() {
  return (
    <ClerkProvider>
      <MultisessionAppSupport>
        <Page />
      </MultisessionAppSupport>
    </ClerkProvider>
  );
}

function Page() {
  useEffect(() => {
    // no need to check if window.Clerk exists
    document.title = 'This page uses Clerk ' + window.Clerk.version;
  }, []);

  return <div>The content </div>;
}

export default App;

```
