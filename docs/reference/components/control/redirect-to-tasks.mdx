The `<RedirectToTasks />` component will navigate users to the appropriate task page when they have pending session tasks.

## Example

<If sdk="nextjs">
  ```tsx {{ filename: 'app/page.tsx' }}
    import { SignedOut, RedirectToTasks } from '@clerk/nextjs'

    export default function Layout({ children }: { children: React.ReactNode }) {
        return (
            <>
            <SignedOut>
                <RedirectToTasks />
            </SignedOut>
            {children}
            </>
        )
    }
  ```
</If>

<If sdk="react">
  ```tsx {{ filename: 'pages/index.tsx' }}
  import { SignedOut, RedirectToTasks } from '@clerk/clerk-react'

  export default function Page() {
    return (
      <>
        <SignedOut>
          <RedirectToTasks />
        </SignedOut>
      </>
    )
  }
  ```
</If>

<If sdk="react-router">
  ```tsx {{ filename: 'app/routes/home.tsx' }}
  import { SignedIn, SignedOut, RedirectToTasks } from '@clerk/react-router'

  export default function Home() {
    return (
      <>
  ```
