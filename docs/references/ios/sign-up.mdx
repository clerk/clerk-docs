### Create a new Sign Up

First, let's create a `SignUpView`. This view will handle collecting the email address and password from the user.
This view will also have a function to create the `SignUp` via the Clerk API.

We'll host this view inside a container view called `MainView`

<CodeBlockTabs options={["SignUpView.swift", "MainView.swift"]}>

```swift filename="SignUpView.swift"
import SwiftUI
// 1. Import the ClerkSDK
import ClerkSDK

struct SignUpView: View {
    // 2. Create a reference to the shared Clerk object
    @ObservedObject private var clerk = Clerk.shared
    // 3. Collect the email address and password from the user
    @State private var emailAddress = ""
    @State private var password = ""
    
    var body: some View {
        VStack {
            TextField("Email Address", text: $emailAddress)
            SecureField("Password", text: $password)
            Button("Sign Up", action: signUp)
        }
        .padding()
    }
    
    // 4. Create the new sign up object via the Clerk API
    private func signUp() {
        Task {
            do {
                try await clerk.client?.createSignUp(.standard(emailAddress: emailAddress, password: password))
            } catch {
                print(error.localizedDescription)
            }
        }
    }
}
```

```swift filename="MainView.swift"
import SwiftUI

struct MainView: View {
    
    var body: some View {
        ScrollView {
            SignUpView()
        }
    }
}
```

### Verify the Email Address

<Callout>
Depending on the settings you choose in your dashboard, you may or may not require new users to verify their email address before signing up. 
We strongly recommend doing so and will demonstrate the verification process here.
</Callout>

After successfully creating the sign up, if we inspect the new `SignUp` object, we'll notice we have a `status` of `"missing_requirements"` and the `unverified_fields` contains a value of `"email_address"`.

```json
"object": "sign_up_attempt",
"id": "sua_2fElBiYiPkw2Ntk6GACCVPx6sPg",
"status": "missing_requirements",
"required_fields": [...],
"optional_fields": [...],
"missing_fields": [...],
"unverified_fields": [
    "email_address"
]
```  

This indicates that in order to continue the sign up, we need to verify the user's email address first. Let's add some code to our `SignUpView` to do that...

<CodeBlockTabs options={["SignUpView.swift"]}>

```swift filename="SignUpView.swift"
import SwiftUI
import ClerkSDK

struct SignUpView: View {
    @ObservedObject private var clerk = Clerk.shared
    @State private var emailAddress = ""
    @State private var password = ""
    // 1. Collect the one time pass (OTP) code
    @State private var otpCode = ""
    
    var body: some View {
        VStack {
            // 2. Check to see if we already have an active sign up and need to verify the email address.
            if let signUp = clerk.client?.signUp, signUp.unverifiedFields.contains("email_address") {
                TextField("Verification Code", text: $otpCode)
                    // 3. Send the verification code to the email address.
                    .task { prepareVerification() }
                Button("Verify", action: attemptVerification)
            } else {
                TextField("Email", text: $emailAddress)
                SecureField("Password", text: $password)
                Button("Sign Up", action: signUp)
            }
        }
        .padding()
    }
    
    private func signUp() {
        Task {
            do {
                try await clerk.client?.createSignUp(.standard(emailAddress: emailAddress, password: password))
            } catch {
                print(error.localizedDescription)
            }
        }
    }

    // 4. Send a one time pass code (OTP) to the email address used to sign up.
    private func prepareVerification() {
        Task {
            do {
                try await clerk.client?.signUp?.prepareVerification(.emailCode)
            } catch {
                print(error.localizedDescription)
            }
        }
    }
    
    // 5. Attempt to verify the email address by submitting the code input by the user.
    private func attemptVerification() {
        Task {
            do {
                try await clerk.client?.signUp?.attemptVerification(.emailCode(code: otpCode))
            } catch {
                print(error.localizedDescription)
            }
        }
    }
}
```

</CodeBlockTabs>

### Display the new user

Once we have verified all `unverified_fields` and have completed the sign up, we can access our new user via `clerk.user`.

Let's create a new `UserView` to display some information about a user...

<CodeBlockTabs options={["UserView.swift"]}>

```swift filename="UserView.swift"
import SwiftUI
// 1. Import the ClerkSDK
import ClerkSDK

struct UserView: View {
    // 2. Create a parameter for the User object
    let user: User
    
    var body: some View {
        VStack {
            // 3. Display some info about the user
            AsyncImage(url: URL(string: user.imageUrl)) { image in
                image.resizable().scaledToFill()
            } placeholder: {
                Color.secondary
            }
            .frame(width: 50, height: 50)
            .clipShape(.circle)
            
            Text(user.id)
        }
        .padding()
    }
}
```

</CodeBlockTabs>

In our `MainView` that we created back in Step 1, we'll pass the `User` to our `UserView`.

<CodeBlockTabs options={["MainView.swift"]}>
```swift filename="MainView.swift"
import SwiftUI
// 1. Import the ClerkSDK
import ClerkSDK

struct ContentView: View {
    // 2. Create a reference to the shared Clerk object
    @ObservedObject private var clerk = Clerk.shared
    
    var body: some View {
        ScrollView {
            // 3. If we have a user, display the UserView
            if let user = clerk.user {
                UserView(user: user)
            } else {
                SignUpView()
            }
        }
    }
}
```
